CREATE OR REPLACE FUNCTION random_string(length INT) RETURNS TEXT AS $$
DECLARE
    chars TEXT := '0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz';
    result TEXT := '';
    i INT := 0;
BEGIN
    FOR i IN 1..length LOOP
        result := result || substr(chars, (random()*61)::int + 1, 1);
    END LOOP;
    RETURN result;
END;
$$ LANGUAGE plpgsql;

CREATE OR ALTER PROCEDURE dbo.InsertWithUniqueRID
    @Payload NVARCHAR(MAX)
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @NewRID NVARCHAR(20);
    DECLARE @Exists INT = 1;

    -- Loop until a unique ID is generated
    WHILE @Exists > 0
    BEGIN
        SET @NewRID = 'RID_' + LOWER(SUBSTRING(CONVERT(NVARCHAR(36), NEWID()), 1, 15));
        
        SELECT @Exists = COUNT(*) 
        FROM dbo.YourTableName 
        WHERE RecordID = @NewRID;
    END

    -- Final Insert
    INSERT INTO dbo.YourTableName (RecordID, Payload)
    VALUES (@NewRID, @Payload);
    
    -- Return the ID to the application (useful for Streamlit)
    SELECT @NewRID AS GeneratedID;
END;
GO

CREATE OR REPLACE FUNCTION set_updated_at_ignore_columns()
RETURNS TRIGGER AS $$
BEGIN
    -- Only update timestamp if meaningful columns changed
    IF
        NEW IS DISTINCT FROM OLD
        AND (
            NEW.name        IS DISTINCT FROM OLD.name OR
            NEW.email       IS DISTINCT FROM OLD.email OR
            NEW.role        IS DISTINCT FROM OLD.role
        )
    THEN
        NEW.updated_at = NOW();
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DO $$
DECLARE
    r RECORD;
    tables TEXT[] := ARRAY[
        'api_users',
        'orders',
        'payments'
    ];
    col_exists BOOLEAN;
BEGIN
    FOREACH r IN ARRAY tables LOOP

        -- Check if 'updated_at' column exists
        SELECT EXISTS (
            SELECT 1
            FROM information_schema.columns
            WHERE table_schema = 'public'
              AND table_name = r
              AND column_name = 'updated_at'
        ) INTO col_exists;

        IF col_exists THEN
            -- Drop existing trigger (idempotent)
            EXECUTE format(
                'DROP TRIGGER IF EXISTS trg_%I_updated_at ON public.%I',
                r, r
            );

            -- Create trigger
            EXECUTE format(
                'CREATE TRIGGER trg_%I_updated_at
                 BEFORE UPDATE ON public.%I
                 FOR EACH ROW
                 EXECUTE FUNCTION set_updated_at_timestamp()',
                r, r
            );

            RAISE NOTICE 'Trigger created on table: %', r;
        ELSE
            RAISE NOTICE 'Skipping table % because updated_at does not exist', r;
        END IF;

    END LOOP;
END $$;
